{{- /* Generate dynamic OG image SVG for each page */ -}} {{- $title := .Title | default .Site.Title -}} {{- $description := or .Description (.Summary | plainify | truncate 120) .Site.Params.description -}} {{- $bgColor := "#00647c" -}} {{- $textColor := "#ffffff" -}} {{- $subtitleColor := "#b0e0e6" -}} {{- /* Truncate title to fit SVG */ -}} {{- if gt (len $title) 60 -}} {{- $title = printf "%s…" (substr $title 0 57) -}} {{- end -}} {{- /* Truncate description */ -}} {{- if gt (len $description) 100 -}} {{- $description = printf "%s…" (substr $description 0 97) -}} {{- end -}} {{- $svg := printf `` $subtitleColor $textColor $title $subtitleColor $description $textColor $subtitleColor $subtitleColor -}} {{- /* Create resource from string */ -}} {{- $ogImage := resources.FromString (printf "og-images/%s.svg" .File.Path) $svg -}} {{- /* Return the permalink */ -}} {{- $ogImage.Permalink -}}
